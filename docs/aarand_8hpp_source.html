33template<typename Output_ = double, class Engine_>
34Output_ standard_uniform(Engine_& eng) {
35 typedef typename Engine_::result_type R;
36 static_assert(std::numeric_limits<R>::is_integer, "RNG engine must yield integer results");
37
38 // Can't be bothered to figure out whether the range fits into 'R' for signed values.
39 // So instead, we just require unsigned integers, where the range will always fit.
40 static_assert(!std::numeric_limits<R>::is_signed, "RNG engine must yield unsigned integers");
41
42 // Stolen from Boost, see https://www.boost.org/doc/libs/1_67_0/boost/random/uniform_01.hpp
43 // The +1 probably doesn't matter for 64-bit generators, but is helpful for engines with
44 // fewer output bits, to reduce the (small) probability of sampling 1's.
45 constexpr Output_ ONE = 1;
46 constexpr Output_ factor = ONE / (static_cast<Output_>(Engine_::max() - Engine_::min()) + ONE);
47
48 // Note that it still might be possible to get a result = 1, depending on
49 // the numerical precision used to compute the product; hence the loop.
50 Output_ result;
51 do {
52 result = static_cast<Output_>(eng() - Engine_::min()) * factor;
53 } while (result == ONE);
54
55 return result;
56}

61// Some of the functions below log-transform a uniform random variable.
62// However, standard_uniform() has a small chance of returning zero, resulting in an undesirable -Inf after log.
63// To avoid this, any time we sample zero, we pretend that we sampled 1 instead.
64// This is equivalent to "wrapping around" the sampling interval so that 0 "underflows" to 1, i.e., `[0, 1)` becomes `(0, 1]`.
65// The log-transform of 1 is subsequently zero, which is much nicer.
66template<typename Output_>
67Output_ wrapped_log(const Output_ val) {
68 return (val == 0 ? static_cast<Output_>(0) : std::log(val));
69}
84template<typename Output_ = double, class Engine_>
85std::pair<Output_, Output_> standard_normal(Engine_& eng) {
86 constexpr Output_ PI = 3.14159265358979323846;
87 constexpr Output_ TWO = 2;
88
89 // Box-Muller gives us two random values at a time.
90 const Output_ constant = std::sqrt(-TWO * wrapped_log(standard_uniform<Output_>(eng)));
91 const Output_ angle = TWO * PI * standard_uniform<Output_>(eng);
92 return std::make_pair(constant * std::sin(angle), constant * std::cos(angle));
93}

121template<typename Output_ = int, class Engine_>
122Output_ discrete_uniform(Engine_& eng, const Output_ bound) {
123 static_assert(std::numeric_limits<Output_>::is_integer);
124 if (bound <= 0) {
125 throw std::runtime_error("'bound' should be a positive integer");
126 }
127
128 typedef typename Engine_::result_type R;
129 static_assert(std::numeric_limits<R>::is_integer);
130 static_assert(!std::numeric_limits<R>::is_signed); // don't want to figure out how to store the range if it might not fit into R.
131
132 constexpr R range = Engine_::max() - Engine_::min();
133 if (static_cast<typename std::make_unsigned<Output_>::type>(bound) > range) { // force an unsigned comparison.
134 throw std::runtime_error("'bound' should be less than the RNG range");
135 }
136
137 R draw = eng() - Engine_::min();
138
139 // Conservative shortcut to avoid an extra modulo operation in computing
140 // 'limit' if 'draw' is below 'limit'. This is based on the observation
141 // that 'range - bound <= limit', so any condition that triggers the loop
142 // will also pass this check. Allows early return when 'range >> bound'.
143 if (draw > range - bound) {
144
145 // The limit is necessary to provide uniformity in the presence of the
146 // modulus. The idea is to re-sample if we get a draw above the limit.
147 // Technically this can have problems as bound approaches range, in which
148 // case we might end up discarding a lot of the sample space... but this
149 // is unlikely to happen in practice, and even if it does, it's a rejection
150 // rate that's guaranteed to be less than 50%, so whatever.
151 //
152 // Note that the +1 is necessary because range is inclusive but bound is not.
153 const R limit = range - ((range % bound) + 1);
154
155 // In addition, we don't have to deal with the crap about combining draws
156 // to get enough entropy, which is 90% of the Boost implementation.
157 while (draw > limit) {
158 draw = eng() - Engine_::min();
159 }
160 }
161
162 return draw % bound;
163}

176template<class InputIterator_, typename Length_, class Engine_>
177void shuffle(const InputIterator_ values, const Length_ n, Engine_& eng) {
178 if (n <= 1) {
179 return;
180 }
181
182 const Length_ last = n - 1;
183 for (Length_ i = 0; i < last; ++i) {
184 const auto chosen = discrete_uniform(eng, n - i);
185 if (chosen) {
186 const auto current = values + i;
187 using std::swap;
188 swap(*current, *(current + chosen));
189 }
190 }
191}

208template<class InputIterator_, typename Length_, class OutputIterator_, class Engine_>
209void sample(InputIterator_ values, const Length_ n, const Length_ s, OutputIterator_ output, Engine_& eng) {
210 if (!s) {
211 return;
212 }
213
214 auto remaining = s;
215 for (Length_ i = 0; i < n; ++i, ++values) {
216 const Length_ denom = n - i;
217 const double threshold = static_cast<double>(remaining) / denom;
218 if (threshold >= 1) {
219 // Once remaining >= denom, all remaining values must be selected.
220 // Both values will drop at the same rate so threshold will always be >= 1 in subsequent loops.
221 std::copy_n(values, denom, output);
222 return;
223 }
224
225 if (standard_uniform(eng) <= threshold) {
226 *output = *values;
227 ++output;
228 --remaining;
229 if (!remaining) {
230 return;
231 }
232 }
233 }
234}

250template<typename Length_, class OutputIterator_, class Engine_>
251void sample(const Length_ bound, const Length_ s, OutputIterator_ output, Engine_& eng) {
252 if (!s) {
253 return;
254 }
255
256 auto remaining = s;
257 for (Length_ i = 0; i < bound; ++i) {
258 const Length_ denom = bound - i;

260 if (threshold >= 1) {
261 // Once remaining >= denom, all remaining indices must be selected.
262 // Both values will drop at the same rate so threshold will always be >= 1 in subsequent loops.
263 std::iota(output, output + denom, i);
264 return;
265 }
266
267 if (standard_uniform(eng) <= threshold) {
268 *output = i;
