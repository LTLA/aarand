26template<typename T = double, class Engine>
27T standard_uniform(Engine& eng) {
28 typedef typename Engine::result_type R;
29 static_assert(std::numeric_limits<R>::is_integer, "RNG engine must yield integer results");
30
31 // Can't be bothered to figure out whether the range fits into 'R' for signed values.
32 // So instead, we just require unsigned integers, where the range will always fit.
33 static_assert(!std::numeric_limits<R>::is_signed, "RNG engine must yield unsigned integers");
34
35 // Make sure we get the right type to avoid inadvertent promotions.
36 constexpr T ONE_ = 1;
37
38 // Stolen from Boost, see https://www.boost.org/doc/libs/1_67_0/boost/random/uniform_01.hpp
39 // The +1 probably doesn't matter for 64-bit generators, but is helpful for engines with
40 // fewer output bits, to reduce the (small) probability of sampling 1's.
41 constexpr T factor = ONE_ / (static_cast<T>(Engine::max() - Engine::min()) + ONE_);
42
43 // Note that it still might be possible to get a result = 1, depending on
44 // the numerical precision used to compute the product; hence the loop.
45 T result;
46 do {
47 result = static_cast<T>(eng() - Engine::min()) * factor;
48 } while (result == ONE_);
49
50 return result;
51}

63template<typename T = double, class Engine>
64std::pair<T, T> standard_normal(Engine& eng) {
65 constexpr T PI_ = 3.14159265358979323846;
66 constexpr T TWO_ = 2;
67
68 // Box-Muller gives us two random values at a time.
69 T constant = std::sqrt(-TWO_ * std::log(standard_uniform<T>(eng)));
70 T angle = TWO_ * PI_ * standard_uniform<T>(eng);
71 return std::make_pair(constant * std::sin(angle), constant * std::cos(angle));
72}

103template<typename T = int, class Engine>
104T discrete_uniform(Engine& eng, T bound) {
105 typedef typename Engine::result_type R;
106 static_assert(std::numeric_limits<R>::is_integer);
107 static_assert(!std::numeric_limits<R>::is_signed); // don't want to figure out how to store the range.
108
109 constexpr R range = Engine::max() - Engine::min();
110 if (bound > range) {
111 throw std::runtime_error("'bound' should be less than the RNG range");
112 }
113
114 static_assert(std::numeric_limits<T>::is_integer);
115 if (bound <= 0) {
116 throw std::runtime_error("'bound' should be a positive integer");
117 }
118
119 R draw = eng() - Engine::min();
120
121 // Conservative shortcut to avoid an extra modulo operation in computing
122 // 'limit' if 'draw' is below 'limit'. This is based on the observation
123 // that 'range - bound <= limit', so any condition that triggers the loop
124 // will also pass this check. Allows early return when 'range >> bound'.
125 if (draw > range - bound) {
126
127 // The limit is necessary to provide uniformity in the presence of the
128 // modulus. The idea is to re-sample if we get a draw above the limit.
129 // Technically this can have problems as bound approaches range, in which
130 // case we might end up discarding a lot of the sample space... but this
131 // is unlikely to happen in practice, and even if it does, it's a rejection
132 // rate that's guaranteed to be less than 50%, so whatever.
133 //
134 // Note that the +1 is necessary because range is inclusive but bound is not.
135 const R limit = range - ((range % bound) + 1);
136
137 // In addition, we don't have to deal with the crap about combining draws
138 // to get enough entropy, which is 90% of the Boost implementation.
139 while (draw > limit) {
140 draw = eng() - Engine::min();
141 }
142 }
143
144 return draw % bound;
145}

160 if (n) {
161 using std::swap;
162 for (size_t i = 0; i < n - 1; ++i) {

186template<class In, class Out, class Engine>
187void sample(In values, size_t n, size_t s, Out output, Engine& eng) {
188 for (size_t i = 0; i < n && s; ++i, ++values) {
189 const double threshold = static_cast<double>(s)/(n - i);

191 *output = *values;
