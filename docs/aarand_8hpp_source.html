32template<typename Output_ = double, class Engine_>
33Output_ standard_uniform(Engine_& eng) {
34 typedef typename Engine_::result_type R;
35 static_assert(std::numeric_limits<R>::is_integer, "RNG engine must yield integer results");
36
37 // Can't be bothered to figure out whether the range fits into 'R' for signed values.
38 // So instead, we just require unsigned integers, where the range will always fit.
39 static_assert(!std::numeric_limits<R>::is_signed, "RNG engine must yield unsigned integers");
40
41 // Stolen from Boost, see https://www.boost.org/doc/libs/1_67_0/boost/random/uniform_01.hpp
42 // The +1 probably doesn't matter for 64-bit generators, but is helpful for engines with
43 // fewer output bits, to reduce the (small) probability of sampling 1's.
44 constexpr Output_ ONE = 1;
45 constexpr Output_ factor = ONE / (static_cast<Output_>(Engine_::max() - Engine_::min()) + ONE);
46
47 // Note that it still might be possible to get a result = 1, depending on
48 // the numerical precision used to compute the product; hence the loop.
49 Output_ result;
50 do {
51 result = static_cast<Output_>(eng() - Engine_::min()) * factor;
52 } while (result == ONE);
53
54 return result;
55}

67template<typename Output_ = double, class Engine_>
68std::pair<Output_, Output_> standard_normal(Engine_& eng) {
69 constexpr Output_ PI = 3.14159265358979323846;
70 constexpr Output_ TWO = 2;
71
72 // Box-Muller gives us two random values at a time.
73 const Output_ constant = std::sqrt(-TWO * std::log(standard_uniform<Output_>(eng)));
74 const Output_ angle = TWO * PI * standard_uniform<Output_>(eng);
75 return std::make_pair(constant * std::sin(angle), constant * std::cos(angle));
76}

107template<typename Output_ = int, class Engine_>
108Output_ discrete_uniform(Engine_& eng, const Output_ bound) {
109 static_assert(std::numeric_limits<Output_>::is_integer);
110 if (bound <= 0) {
111 throw std::runtime_error("'bound' should be a positive integer");
112 }
113
114 typedef typename Engine_::result_type R;
115 static_assert(std::numeric_limits<R>::is_integer);
116 static_assert(!std::numeric_limits<R>::is_signed); // don't want to figure out how to store the range if it might not fit into R.
117
118 constexpr R range = Engine_::max() - Engine_::min();
119 if (static_cast<typename std::make_unsigned<Output_>::type>(bound) > range) { // force an unsigned comparison.
120 throw std::runtime_error("'bound' should be less than the RNG range");
121 }
122
123 R draw = eng() - Engine_::min();
124
125 // Conservative shortcut to avoid an extra modulo operation in computing
126 // 'limit' if 'draw' is below 'limit'. This is based on the observation
127 // that 'range - bound <= limit', so any condition that triggers the loop
128 // will also pass this check. Allows early return when 'range >> bound'.
129 if (draw > range - bound) {
130
131 // The limit is necessary to provide uniformity in the presence of the
132 // modulus. The idea is to re-sample if we get a draw above the limit.
133 // Technically this can have problems as bound approaches range, in which
134 // case we might end up discarding a lot of the sample space... but this
135 // is unlikely to happen in practice, and even if it does, it's a rejection
136 // rate that's guaranteed to be less than 50%, so whatever.
137 //
138 // Note that the +1 is necessary because range is inclusive but bound is not.
139 const R limit = range - ((range % bound) + 1);
140
141 // In addition, we don't have to deal with the crap about combining draws
142 // to get enough entropy, which is 90% of the Boost implementation.
143 while (draw > limit) {
144 draw = eng() - Engine_::min();
145 }
146 }
147
148 return draw % bound;
149}

190template<class InputIterator_, typename Length_, class OutputIterator_, class Engine_>
191void sample(InputIterator_ values, const Length_ n, const Length_ s, OutputIterator_ output, Engine_& eng) {
192 if (!s) {
193 return;
194 }
195
196 auto remaining = s;
197 for (Length_ i = 0; i < n; ++i, ++values) {
198 const Length_ denom = n - i;
199 const double threshold = static_cast<double>(remaining) / denom;
200 if (threshold >= 1) {
201 // Once remaining >= denom, all remaining values must be selected.
202 // Both values will drop at the same rate so threshold will always be >= 1 in subsequent loops.
203 std::copy_n(values, denom, output);
204 return;
205 }
206
207 if (standard_uniform(eng) <= threshold) {
208 *output = *values;
209 ++output;
210 --remaining;
211 if (!remaining) {
212 return;
213 }
214 }
215 }
216}
