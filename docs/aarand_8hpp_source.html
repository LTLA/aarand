25template<typename T = double, class Engine>
26T standard_uniform(Engine& eng) {
27 typedef typename Engine::result_type R;
28 static_assert(std::numeric_limits<R>::is_integer);
29
30 // Can't be bothered to figure out whether the range fits into 'R' for signed values.
31 // So instead, we just require unsigned integers, where the range will always fit.
32 static_assert(!std::numeric_limits<R>::is_signed);
33
34 // Stolen from Boost, see https://www.boost.org/doc/libs/1_67_0/boost/random/uniform_01.hpp
35 // The +1 probably doesn't matter for 64-bit generators, but is helpful for engines with
36 // fewer output bits, to reduce the (small) probability of sampling 1's.
37 constexpr double factor = 1.0 / (static_cast<T>(Engine::max() - Engine::min()) + 1.0);
38
39 // Note that it still might be possible to get a result = 1, depending on
40 // the numerical precision used to compute the product; hence the loop.
41 double result;
42 do {
43 result = static_cast<T>(eng() - Engine::min()) * factor;
44 } while (result == 1.0);
45 return result;
46}

57template<typename T = double, class Engine>
58std::pair<T, T> standard_normal(Engine& eng) {
59 constexpr double pi = 3.14159265358979323846;
60
61 // Box-Muller gives us two random values at a time.
62 double constant = std::sqrt(-2 * std::log(standard_uniform<T>(eng)));
63 double angle = 2 * pi * standard_uniform<T>(eng);
64 return std::make_pair(constant * std::sin(angle), constant * std::cos(angle));
65}

95template<typename T = int, class Engine>
96T discrete_uniform(Engine& eng, T bound) {
97 typedef typename Engine::result_type R;
98 static_assert(std::numeric_limits<R>::is_integer);
99 static_assert(!std::numeric_limits<R>::is_signed); // don't want to figure out how to store the range.
100
101 constexpr R range = Engine::max() - Engine::min();
102 if (bound > range) {
103 throw std::runtime_error("'bound' should be less than the RNG range");
104 }
105
106 static_assert(std::numeric_limits<T>::is_integer);
107 if (bound <= 0) {
108 throw std::runtime_error("'bound' should be a positive integer");
109 }
110
111 // The limit is necessary to provide uniformity in the presence of the
112 // modulus. The idea is to re-sample if we get a draw above the limit.
113 // Technically this can have problems as bound approaches range, in which
114 // case we might end up discarding a lot of the sample space... but this
115 // is unlikely to happen in practice, so whatever. Note that the +1 is
116 // necessary because range is inclusive but bound is not.
117 const R limit = range - (range % bound + 1);
118
119 // In addition, we don't have to deal with the crap about combining draws
120 // to get enough entropy, which is 90% of the Boost implementation.
121 R draw;
122 do {
123 draw = eng() - Engine::min();
124 } while (draw > limit);
125
126 return draw % bound;
127}

168template<class In, class Out, class Engine>
169void sample(In values, size_t n, size_t s, Out output, Engine& eng) {
170 for (size_t i = 0; i < n && s; ++i, ++values) {
171 const double threshold = static_cast<double>(s)/(n - i);
172 if (threshold >= 1 || standard_uniform(eng) <= threshold) {
173 *output = *values;
174 ++output;
175 --s;
176 }
